Auton Embeddable Widget Example

Copy the code below to embed the Auton tip button on your website:

<!-- Auton Tip Button Widget -->
<script>
  (function() {
    var autonTipButton = document.createElement('a');
    autonTipButton.className = 'auton-tip-button';
    autonTipButton.href = 'YOUR_TIP_LINK_HERE';
    autonTipButton.target = '_blank';
    autonTipButton.rel = 'noopener noreferrer';
    autonTipButton.innerHTML = '<svg class="auton-tip-icon" viewBox="0 0 24 24" fill="currentColor"><path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2zm-2 15l-5-5 1.41-1.41L10 14.17l7.59-7.59L19 8l-9 9z"/></svg> Tip with Auton';
    
    // Insert the button into the page
    var script = document.currentScript;
    script.parentNode.insertBefore(autonTipButton, script);
  })();
</script>
